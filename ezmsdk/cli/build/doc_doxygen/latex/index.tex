\begin{DoxyCode}{0}
\DoxyCodeLine{mkdir build}
\DoxyCodeLine{cd build}
\DoxyCodeLine{cmake ..}
\DoxyCodeLine{cmake -\/-\/build .}
\end{DoxyCode}


go to bin folder to run the application or run the unit test
